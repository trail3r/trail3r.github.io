#### 긴 일반 코드 블럭
```python
from tensorflow import keras
from sklearn.model_selection import train_test_split
import matplotlib.pyplot as plt
import numpy as np

# 데이터셋 저장
(train_input, train_target), (test_input, test_target) = keras.datasets.fashion_mnist.load_data()

# 처음 10개의 이미지 및 레이블 확인
fig, ax = plt.subplots(1, 10)   
for i in range(10):
    ax[i].imshow(train_input[i], cmap='gray_r')   # 흑백 이미지 반전 처리 : 0(white) ~ 1(black)
    ax[i].axis('off')
plt.show()
print('Labels for 10 train dataset = ', [train_target[i] for i in range(10)])

print(train_input.shape)
print(test_input.shape)

# 픽셀 값을 0 ~ 1 사이의 값으로 정규화
train_scaled = train_input / 255.0    
test_scaled = test_input / 255.0     

# 학습용 데이터 6만장의 20%를 검증용 데이터로 배정 : 학습 48,000장, 검증 12,000장, 시험 10,000장
# 랜덤 혼합 후 학습/검증용으로 분리
train_scaled, val_scaled, train_target, val_target = train_test_split(train_scaled, train_target, test_size=0.2)

model = keras.Sequential()                           # 신경망 객체 만들기 
model.add(keras.layers.Flatten(input_shape=(28, 28))) # 입력층 : 28x28 = 784 뉴런 (한줄로 펼치기)
model.add(keras.layers.Dense(500, activation='relu')) # 은닉층(밀집연결, 완전연결) : 100 뉴런
model.add(keras.layers.Dropout(0.3))                  # 드롭아웃(30% 데이터 삭제, overfit 예방) : 70 뉴런
model.add(keras.layers.Dense(300, activation='relu')) # 은닉층(밀집연결, 완전연결) : 100 뉴런
model.add(keras.layers.Dropout(0.3))                  # 드롭아웃(30% 데이터 삭제, overfit 예방) : 70 뉴런
model.add(keras.layers.Dense(200, activation='relu')) # 은닉층(밀집연결, 완전연결) : 100 뉴런
model.add(keras.layers.Dropout(0.3))      
model.add(keras.layers.Dense(10, activation='softmax')) # 출력층 : 10 뉴런

# 이 모델은 2-layer 모델이다.

model.compile(optimizer='adam', loss='sparse_categorical_crossentropy', metrics=['accuracy'])
# 경사하강법(adam), 손실함수(정수 레이블 적용 cross etropy), 측정기준(정확도) 설정

checkpoint_cb = keras.callbacks.ModelCheckpoint('fashion_mnist.h5', save_best_only=True)
# 검증데이터 기준으로 최저 손실에 해당하는 모델을 ‘fashion_mnist’이라는 이름의 h5 형식으로 저장 

early_stopping_cb = keras.callbacks.EarlyStopping(patience=2, restore_best_weights=True)
# patience 횟수만큼 검증 손실이 감소하지 않으면 조기 종료하고 최저 손실에 해당하는 가중치 복원

history = model.fit(train_scaled, train_target, epochs=20, validation_data=(val_scaled, val_target),
           callbacks=[checkpoint_cb, early_stopping_cb])
# 학습 데이터/레이블, 횟수, 검증 데이터/레이블, 콜백 설정(체크포인트/조기종료) 후 학습수행

model.summary()  # 구성한 신경망의 종류/형태/가중치 개수 출력
print(history.history.keys()) # 에폭 별로 loss 및 metrics 값 반환 : 딕셔너리 형식
                              # ['loss', 'accuracy', 'val_loss', 'val_accuracy'])

plt.plot(history.history['loss'])      # 학습 데이터에 대한 손실함수 그래프
plt.plot(history.history['val_loss'])  # 검증 데이터에 대한 손실함수 그래프
plt.xlabel('epoch')                 # x축 이름
plt.ylabel('loss')                   # y축 이름
plt.legend(['train', 'val'])          # 범례
plt.show()

# 체크포인트에서 저장한 모델 불러오기, 모델 저장 경로에 한글이 없어야 함
model = keras.models.load_model('fashion_mnist.h5')
model.evaluate(test_scaled, test_target)
# 시험용 데이터에 학습한 모델을 적용하여 평가 : 손실 및 정확도 반환

# 시험 이미지에 대한 추론 예시
i = 65
new_data = test_scaled[i:i+2]   # shape (-1, 28, 28) 형태로 새로운 데이터 사용 | 이미지 2개를 뽑아내는 거임
p = model.predict(new_data)
print(p.round(2))              # 예측 확률
print(np.argmax(p, axis=1))   # 예측 레이블
print(test_target[i:i+2])        # 실제 레이블
```
